    <script>

        document.addEventListener('DOMContentLoaded', function () {

            // Get all necessary elements from the DOM

            const slides = document.querySelectorAll('.slide');

            const prevBtn = document.getElementById('prev-btn');

            const nextBtn = document.getElementById('next-btn');

            const slideCounter = document.getElementById('slide-counter');

            

            // Set initial state

            let currentSlide = 1;

            const totalSlides = slides.length;


            // Function to update the view based on the current slide

            function updateView() {

                // Loop through all slides to set their visibility

                slides.forEach(slide => {

                    const slideNumber = parseInt(slide.getAttribute('data-slide'));

                    if (slideNumber === currentSlide) {

                        slide.classList.remove('slide-hidden');

                        slide.classList.add('slide-active');

                    } else {

                        slide.classList.add('slide-hidden');

                        slide.classList.remove('slide-active');

                    }

                });


                // Update the slide counter text

                slideCounter.textContent = `Slide ${currentSlide} of ${totalSlides}`;


                // Enable or disable navigation buttons

                prevBtn.disabled = currentSlide === 1;

                nextBtn.disabled = currentSlide === totalSlides;

            }


            // Event listener for the "Next" button

            nextBtn.addEventListener('click', () => {

                if (currentSlide < totalSlides) {

                    currentSlide++;

                    updateView();

                }

            });


            // Event listener for the "Previous" button

            prevBtn.addEventListener('click', () => {

                if (currentSlide > 1) {

                    currentSlide--;

                    updateView();

                }

            });

            

            // Keyboard navigation

            document.addEventListener('keydown', (e) => {

                if(e.key === 'ArrowRight' && currentSlide < totalSlides) {

                    currentSlide++;

                    updateView();

                } else if(e.key === 'ArrowLeft' && currentSlide > 1) {

                    currentSlide--;

                    updateView();

                }

            });


            // Initialize the view on page load

            updateView();

        });

    </script>
